CMAKE_MINIMUM_REQUIRED (VERSION 2.8.12)

PROJECT (libraw_test)

SET (CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/../../../cmake/Modules/")
FIND_PACKAGE( LibRaw )
IF( HAVE_LIBRAW )
  INCLUDE_DIRECTORIES( ${LIBRAW_INCLUDE_DIR} )
  LINK_DIRECTORIES( ${LIBRAW_LIBRARY_DIRS} )
ELSE(HAVE_LIBRAW)
  MESSAGE( "whoops - no libRaw found" )
ENDIF(HAVE_LIBRAW)

ADD_LIBRARY( libraw_test_plugin
             SHARED libraw_test.cxx )
TARGET_LINK_LIBRARIES ( libraw_test_plugin ${EXTRA_LIBS} ${LIBRAW_LIBRARIES} ${EXIF_LIBRARIES} )
